#Title: Job processing log
#Description: Retrieve job log if available. Start and count GET params to define the line numbers you want to retrieve. By default the first 300 logs are sent but can be altered using the "start" and "count" keys. If count is zero (0) all available job logs from "start" are returned.

#Response: 200 OK
#Content-Type:application/json; charset=utf-8
#Body:JSON

#Title: Submit job
#Description: Submit a file to a workflow. "start_proc" is optional but lets you choose a spesific node to submit the file to. The "Variables" are optional but lets you inject data to the specified variables used by the workflow. Available user variables are obtained from /workflows/<workflow id>/user_variables.
#Method: POST
#Content-Type:application/json
#URI: http://127.0.0.1:port/path/jobs/<job_id>
#Body:JSON
{

"wf_id": "<workflow id>",
"inputfile": "<full path to file>",
"start_proc": "<processor node id>",
"priority": "integer 0-5",
"variables": [

{
"name": "<s_string>",
"data": "<string>"
}, {
"name": "<i_string>",
"data": <integer>
}, {
"name": "<f_string>",
"data": <number>
}

]

}


#Response: 202 Accepted
#Content-Type:application/json; charset=utf-8
#Body:JSON
{

"uri": "<url to the job>",
"job_id": "<identifier for the job>"

}

#Title: Pause, resume and abort sub jobs
#Description: Set the sub job to pause, resume or queue to abort. Allowed states: "pause", "resume" or "abort". "split_id" defines what instance/sub/branch/split id of the job you want to pause/resume/abort. The same method can be used on finished jobs to either "retry" or "retry_fail" (retry at failed node).
#Method: PUT
#Content-Type:application/json
#URI: http://127.0.0.1:port/path/jobs/<job_id>
#Body:JSON
{

"action": "<state>",
"split_id": "<split identifier>"

}

#Response: 202 Accepted
